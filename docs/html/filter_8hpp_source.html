49 template <typename Iter, typename Elem, typename Pred, typename Dest>
50 class filtered_iter {
51  Iter source_iter;
52  Iter source_iter_end;
53  // non-owning pointers
54  Pred* predicate;
55  Dest* offcast_dest;
56  // space in iter for placement of Elem instance (to not require DefaultConstructible)
57  typename std::aligned_storage<sizeof(Elem), std::alignment_of<Elem>::value>::type current_space;
58  Elem* current_ptr = nullptr;
59 
60 public:
61  using value_type = Elem;
62  // does not support input_iterator, but valid tag is required
63  // for std::iterator_traits with libc++ on mac
64  using iterator_category = std::input_iterator_tag;
65  using difference_type = std::nullptr_t;
66  using pointer = std::nullptr_t;
67  using reference = std::nullptr_t;

77  filtered_iter(Iter source_iter, Iter source_iter_end, Pred& predicate, Dest& offcast_dest) :
78  source_iter(std::move(source_iter)),
79  source_iter_end(std::move(source_iter_end)),
80  predicate(&predicate),
81  offcast_dest(&offcast_dest) {
82  if (this->source_iter != this->source_iter_end) {
83  this->current_ptr = new (std::addressof(current_space)) Elem(std::move(*this->source_iter));
84  if (!(*this->predicate)(*current_ptr)) {
85  (*this->offcast_dest)(std::move(*current_ptr));
86  next();
87  }
88  }
89  }

192  void next() {
193  for (++source_iter; source_iter != source_iter_end; ++source_iter) {
194  *current_ptr = std::move(*source_iter);
195  auto& ref = *current_ptr;
